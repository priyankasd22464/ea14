    const Calculator = (A, B, C) =>  {
        
           switch (A) {
    
            case "+":
                return B + C;
    
            case "-":
                return B - C;
    
            case "*":
                return B * C;
    
            case "/":
                return B / C;
    
            default:
                return 'return result';
        }
    };
